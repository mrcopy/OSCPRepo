#!/bin/sh

#Obtain a list of ciphers available on a machine
#And attempt to decrypt them with a key
#IF YOU HAVE THE CORRECT KEY
#it will tell you which cipher was used to encrypt it
#
#-pass can also be a file:/<file>
#

ciphers=`openssl list-cipher-commands`
key="INSERT ENCRYPTION KEY"
enc_file="INSERT FILE"

for i in $ciphers; do
 openssl enc -d -${i} -in $enc_file -k $key > /dev/null 2>&1
 if [[ $? -eq 0 ]]; then
  echo "Cipher is $i: openssl enc -d -${i} -in $enc_file -k $key -out foo"
  exit
 fi
done
